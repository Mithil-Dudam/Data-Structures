class HashMap:
    def __init__(self, capacity):
        self.capacity = capacity 
        self.size = 0
        self.buckets = [[] for _ in range(capacity)]

    def length(self):
        return self.size 

    def contains(self, key):
        index = self._hash_function(key)
        bucket = self.buckets[index]
        for k, v in bucket:
            if k == key:
                return True
        return False

    def put(self, key, value):
        index = self._hash_function(key)
        bucket = self.buckets[index]
        for i, (k, v) in enumerate(bucket):
            if k == key:
                bucket[i] = (key, value)
                return 
        bucket.append((key, value))
        self.size += 1

    def get(self, key):
        index = self._hash_function(key)
        bucket = self.buckets[index]
        for k, v in bucket:
            if k == key:
                return v 
        return "Key not found"

    def remove(self, key):
        index = self._hash_function(key)
        bucket = self.buckets[index]
        for i, (k, v) in enumerate(bucket):
            if k == key:
                del bucket[i]
                self.size -= 1
                return
        return "Key not found"
    
    def keys(self):
        keys = []
        for bucket in self.buckets:
            for k, _ in bucket:
                keys.append[k]

    
    def values(self):
        values = []
        for bucket in self.buckets:
            for _, v in bucket:
                values.append[v] 
    
    def items(self):
        items = []
        for bucket in self.buckets:
            for k, v in bucket:
                items.append[(k, v)]
    
    def _hash_function(self, key):
        key_string = str(key)
        hash_result = 0 
        for char in key_string:
            hash_result = (hash_result * 31 + ord(char)) % self.capacity
        return hash_result
